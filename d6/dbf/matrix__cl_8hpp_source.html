304  require_not_t<is_eigen_contiguous_map<Mat>>...>
305  explicit matrix_cl(Mat&& A,
306  matrix_cl_view partial_view = matrix_cl_view::Entire)
307  : rows_(A.rows()), cols_(A.cols()), view_(partial_view) {
308  using Mat_type = std::decay_t<decltype(A.eval())>;
309  if (size() == 0) {
310  return;
311  }
312  if (is_eigen_matrix_or_array<Mat>::value
313  && std::is_lvalue_reference<Mat>::value) {
314  initialize_buffer(A.eval().data());
315  } else {
316  auto* A_heap = new Mat_type(std::move(A.eval()));
317  try {
318  cl::Event e = initialize_buffer(A_heap->data());
319  // We set a callback that will delete the memory once copying is

321  // callback. OpenCL implementation does. So nothing is lost as event
322  // goes out of scope.
323  e.setCallback(CL_COMPLETE, &delete_it<Mat_type>, A_heap);
