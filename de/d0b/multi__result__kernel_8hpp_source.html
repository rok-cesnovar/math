404  if (!any_output) {

410

412  int n_cols = std::get<0>(expressions).x.thread_cols();
413  const char* function = "results_cl.assignment";

415  if (n_rows * n_cols == 0) {
416  return;

419  check_nonnegative(function, "expr.cols()", n_cols);

427  }
428  cl::Kernel& kernel = impl::kernel_;

430 
431  std::set<const operation_cl_base*> generated;
432  impl::set_args(generated, kernel, arg_num, results, expressions);
433 
434  std::vector<cl::Event> events;
435  impl::get_clear_events(events, results, expressions);
436  cl::Event e;

439  kernel.setArg(arg_num++, n_cols);

442  int wgs_cols = (n_cols + local - 1) / local;
