191  inline kernel_parts get_kernel_parts(
192  std::set<const operation_cl_base*>& generated, name_generator& name_gen,
193  const std::string& i, const std::string& j) const {
194  kernel_parts res{};
195  if (generated.count(this) == 0) {
196  this->var_name = name_gen.generate();
197  generated.insert(this);
198  std::string i_arg = i;
199  std::string j_arg = j;
200  derived().modify_argument_indices(i_arg, j_arg);
201  std::array<kernel_parts, N> args_parts = index_apply<N>([&](auto... Is) {
202  return std::array<kernel_parts, N>{this->get_arg<Is>().get_kernel_parts(
203  generated, name_gen, i_arg, j_arg)...};
204  });
205  res = std::accumulate(args_parts.begin(), args_parts.end(),
206  kernel_parts{});
207  kernel_parts my_part = index_apply<N>([&](auto... Is) {
208  return this->derived().generate(i, j, this->get_arg<Is>().var_name...);
209  });
210  res += my_part;
211  res.body = res.body_prefix + res.body;
212  res.body_prefix = "";
213  }
214  return res;
